// Tipos de nodos para el Árbol de Sintaxis Abstracta (AST)
typedef enum e_ast_node_type
{
	NODE_COMMAND,		// Un comando simple (ls -l)
	NODE_PIPE,			// Una tubería (cmd1 | cmd2)
	NODE_AND,			// Operador lógico &&
	NODE_OR,			// Operador lógico ||
	NODE_SUBSHELL		// Un comando entre paréntesis ( (cmd) )
}						t_ast_node_type;

// Estructura general para los nodos del AST
typedef struct s_ast_node
{
	t_ast_node_type		type;
	struct s_ast_node	*left;	// Hijo izquierdo
	struct s_ast_node	*right;	// Hijo derecho
	t_command			*command; // Si `type` es NODE_COMMAND, este es el comando simple.
	// Opcional: Si necesitas un token list para subshells:
	// t_token_bonus		*subshell_tokens; 
}						t_ast_node;
